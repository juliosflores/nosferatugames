Conteúdo do `/tmp/_relatorio.py`:

```python
import os, json
from datetime import datetime, timedelta
import urllib.request

SUPABASE_URL = os.environ.get("SUPABASE_URL", "https://jcnncmfbodglvoyytgok.supabase.co")
SUPABASE_KEY = os.environ.get("SUPABASE_KEY", "")
ontem = (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d")
hoje = datetime.now().strftime("%Y-%m-%d")

headers = {
    "apikey": SUPABASE_KEY,
    "Authorization": f"Bearer {SUPABASE_KEY}",
    "Content-Type": "application/json"
}

def fetch(endpoint):
    req = urllib.request.Request(f"{SUPABASE_URL}/rest/v1/{endpoint}", headers=headers)
    with urllib.request.urlopen(req) as r:
        return json.loads(r.read())

vendas = fetch(f"purchase?created_at=gte.{hoje}&select=id,total,status,produto_id")
produtos = fetch("produtos?select=id,nome,preco,disponivel&disponivel=eq.true")
try:
    garimpo = fetch(f"garimpo_resultados?created_at=gte.{hoje}&select=titulo,preco_visto,preco_referencia")
except:
    garimpo = []

print(json.dumps({
    "vendas": vendas,
    "total_produtos": len(produtos),
    "garimpo": garimpo,
}, indent=2))
```
